# 实验报告：TLB容量与访问开销测量实验

>实验环境
>操作系统：wsl2虚拟机 + Ubuntu
>硬件信息：处理器为intel i7 12700H 14内核 20逻辑处理器，内存大小为32GB
>（具体的信息，如L1，L2缓存数量）需要联网查找或者利用软件）

### 1.根据要求编写tlb.c

```c
#include <stdio.h>      // 用于 printf, fprintf
#include <stdlib.h>     // 用于 malloc, free, atoi, exit
#include <unistd.h>     // 用于 sysconf
#include <time.h>       // 用于 clock_gettime (高精度计时)

// 定义纳秒到秒的转换常量
#define BILLION 1000000000L

/*main函数的参数约定：
argc是代表命令参数的计数（程序名也包含）
argv[] 则按顺序存储了指向命令行参数（const char型）的指针*/
int main(int argc, char *argv[]) {
    // 1. 参数检查
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <number_of_pages> <number_of_trials>\n", argv[0]);
        exit(EXIT_FAILURE);
    }

    // 将命令行参数转换为整数
    int num_pages = atoi(argv[1]); // 要访问的页面数量
    int trials = atoi(argv[2]);    // 重复轮数 (iterations)

    if (num_pages <= 0 || trials <= 0) {
        fprintf(stderr, "Error: Invalid arguments. Must be > 0.\n");
        exit(EXIT_FAILURE);
    }

    // 2. 动态获取系统页大小 
    long page_size = sysconf(_SC_PAGESIZE); 

    // 3. 计算步长 (Jump)
    int jump = page_size / sizeof(int); 

    // 4. 分配内存
    long total_ints = (long)num_pages * jump; 
    int *a = (int *)malloc(total_ints * sizeof(int));

    if (a == NULL) {
        perror("malloc failed");
        exit(EXIT_FAILURE);
    }

    // 5. 预热 (Warm-up) & 初始化
    // 也可用calloc顺便在分配内存的时候就初始化
    for (long i = 0; i < num_pages * jump; i += jump) {
        a[i] = 0;
    }

    // 6. 准备计时器变量
    struct timespec start, end;
    
    // ================= 开始计时 =================
    // 使用 CLOCK_MONOTONIC，它不受系统时间修改的影响，适合测量时长
    clock_gettime(CLOCK_MONOTONIC, &start);

    // 7. 核心测试循环
    // 外层：重复多次以平滑误差
    for (int j = 0; j < trials; j++) {
        // 内层：遍历每一个不同的页面
        // 注意：i 每次增加 jump，确保正好落在下一页的起始位置
        for (long i = 0; i < num_pages * jump; i += jump) {
            // 读-改-写 操作
            // 确保编译器不会优化掉这段代码
            a[i] += 1; 
        }
    }

    clock_gettime(CLOCK_MONOTONIC, &end);
    // ================= 结束计时 =================

    // 8. 计算结果
    // 计算总纳秒数
    double time_taken_ns = (end.tv_sec - start.tv_sec) * BILLION + (end.tv_nsec - start.tv_nsec);
    
    // 计算总访问次数 = 页面数 * 重复轮数
    long total_accesses = (long)num_pages * trials;

    // 计算每次访问的平均时间 (纳秒)
    double time_per_access = time_taken_ns / total_accesses;

    // 9. 输出结果
    // 格式：页大小(字节)  页面数  平均时间(纳秒)
    // 提示：你可以注释掉下面的 printf，只保留最后一行数据以便绘图脚本读取
    printf("Page size: %ld bytes\n", page_size);
    printf("Pages: %d, Trials: %d\n", num_pages, trials);
    printf("Average time per access: %.4f ns\n", time_per_access);

    // 10. 清理内存
    free(a);

    return 0;
}
```

### 2.调试这个C程序
我们在linux环境直接手动编译：
```bash
gcc -o tlb tlb.c
```

再在同一个目录下执行这个程序，传入程序所需要的两个参数：
这里选择开辟一个横跨2048小页（每页有4096个字节）的整数数组，将访问的循环定为100000次。

### 3.编写脚本运行这个C程序，以得到大量数据
我们编写一个shell文件名为run.sh，让这个脚本按照（2,4,8,16,32,64,128...)的页面递增序列去运行tlb程序，并将结果写入result.csv文件。

```shell
#!/bin/bash

# 1. 编译你的 C 程序 (确保目录下有 tlb.c)
gcc -o tlb tlb.c

# 2. 准备数据文件，写入标题行
echo "Pages,Time" > results.csv

# 3. 循环测试不同的页面数量
# 这里的数字列表就是你要测试的横坐标点
for i in 2 4 8 16 32 64 128 256 512 1024 2048 4096 8192 
do
    # 打印正在跑哪个测试，让你知道进度
    echo "Running test for $i pages..."
    
    # 4. 运行 C 程序
    # ./tlb $i 100000 -> 运行程序，参数是 页数($i) 和 次数(100000)
    # awk '{print $NF}' -> 这是一个文本处理工具，它只抓取输出结果的“最后一个单词”(也就是时间数值)
    # >> results.csv -> 把抓到的时间追加写入到 csv 文件末尾
    
    # 注意：这里假设你的 C 程序最后一行输出格式是 "Average time ... : 0.45 ns"
    # 如果你的输出格式不同，这里可能抓取不到数字。
    # 建议修改 C 程序，让最后一行只输出纯数字，或者按照下面的方式抓取：
    
    # 获取程序输出的完整一行
    output=$(./tlb $i 500000 | tail -n 1)
    
    # 提取其中的数字部分 (假设数字在倒数第二个单词，比如 "0.45 ns")
    # 如果你的 printf 是 "Average time: 0.45 ns"，数字就是倒数第二列
    time=$(echo $output | awk '{print $(NF-1)}')
    
    # 把 "页数,时间" 写入文件
    echo "$i,$time(ns)" >> results.csv
done

echo "Done! Data saved to results.csv"
```

运行这个脚本：
```bash
./run.sh
```

### 4.可视化
我们编写一个python脚本plot.py，将result.csv的数据画成折线图，脚本如下：
```python
import pandas as pd
import matplotlib.pyplot as plt

# 1. 文件名和数据加载
FILE_NAME = 'results.csv'
try:
    # 尝试使用 pandas 读取 CSV 文件
    df = pd.read_csv(FILE_NAME)
except FileNotFoundError:
    print(f"Error: File '{FILE_NAME}' not found. Make sure to run the tlb test first.")
    exit()
except Exception as e:
    print(f"Error reading file: {e}")
    exit()

# 确保列名正确（与你 run.sh 中设置的标题行一致）
if 'Pages' not in df.columns or 'Time(ns)' not in df.columns:
    print("Error: CSV file must contain 'Pages' and 'Time(ns)' columns.")
    exit()

# 2. 数据处理与分析
# 计算每增加一倍页面数量，时间增加了多少（用于辅助判断拐点）
# Shift(-1) 会把下一行的时间值拿到当前行
df['Next_Time'] = df['Time(ns)'].shift(-1)
# 计算增长率 (下一页时间 / 当前页时间)
# 使用 .iloc[:-1] 排除最后一行（因为最后一行Next_Time是NaN）
df['Growth_Rate'] = df['Next_Time'] / df['Time(ns)']
# 找到增长率最高的点，通常就是TLB失效的拐点
max_growth_index = df['Growth_Rate'].idxmax()
# 获取拐点处的页面数
inflection_point_pages = df.loc[max_growth_index, 'Pages']
# 获取拐点处的平均访问时间
inflection_point_time = df.loc[max_growth_index, 'Time(ns)']


# 3. 绘图可视化
plt.figure(figsize=(10, 6))

# 绘制主曲线
plt.plot(df['Pages'], df['Time(ns)'], marker='o', linestyle='-', color='blue', label='Average Access Time')

# 标记拐点
plt.scatter(inflection_point_pages, inflection_point_time, color='red', zorder=5, 
            label=f'Inferred TLB Capacity: {inflection_point_pages} pages')

# 绘制辅助线（可选）
plt.axvline(x=inflection_point_pages * 2, color='red', linestyle='--', alpha=0.5, 
            label=f'Failure Point: {inflection_point_pages * 2} pages')

# 设置图表属性
plt.title('TLB Capacity Measurement: Pages vs. Average Access Time')
plt.xlabel('Number of Pages Accessed (Log Scale)', fontsize=12)
plt.ylabel('Average Access Time (Nanoseconds)', fontsize=12)

# TLB 容量通常是 2 的幂次方，使用 Log 刻度更易观察
plt.xscale('log', base=2) 
plt.grid(True, which="both", linestyle='--', linewidth=0.5)

# 确保图例可见
plt.legend()

# 标记文本
plt.text(inflection_point_pages, inflection_point_time * 1.5, 
         f'TLB Capacity ≈ {int(inflection_point_pages)} Pages', 
         color='red', fontsize=10, ha='center')

# 4. 展示图表或保存
plt.tight_layout() # 自动调整布局，防止标签重叠
plt.show()

# 5. 输出分析结论
print("\n--- Analysis Conclusion ---")
print(f"The maximum jump in access time occurs between accessing {inflection_point_pages} pages and {inflection_point_pages * 2} pages.")
print(f"This indicates the TLB (Translation Lookaside Buffer) capacity is approximately: {int(inflection_point_pages)} entries.")
print("The significant performance drop (拐点) is clearly visible on the graph.")
```

### 四、实验结果分析（结合硬件配置）

本实验运行于 Intel Core i7-12700H 处理器与 WSL2 虚拟化环境之上。该处理器采用现代多级缓存与多级 TLB 设计，并配合复杂的硬件预取、乱序执行及页表行走缓存机制；同时，WSL2 环境引入了二级地址转换，使得一次内存访问可能涉及多层页表与 TLB 缓存。

从实验结果的折线图可以看出，平均访问时间随访问页面数量整体呈上升趋势，但未出现单一、陡峭的性能拐点，而是表现为多个区间内斜率逐步变化的形态。

在页面数量较小时，平均访问时间出现短暂下降，主要原因在于循环控制、分支预测和流水线填充等固定开销在极小工作集下占比较高，随着访问规模略微增大，这些固定成本被摊薄，从而导致单位访问时间下降。该区间不宜用于分析 TLB 行为。

当访问页面数量达到中等规模后，访问延迟开始明显上升，但增长过程相对平缓。这一现象反映了现代处理器多级 TLB 的协同作用：随着 L1 TLB 命中率逐步下降，更高层级的 TLB 及页表行走缓存仍能有效降低地址转换开销，使得性能退化呈现为连续变化而非突变。

在访问页面数量进一步增大时，访问延迟继续上升，说明多级 TLB 及相关缓存逐渐无法容纳全部活跃页表项，页表行走在整体访问中所占比例不断提高。在 WSL2 环境下，二级页表转换机制进一步平滑了这一过程。

综上所述，本实验结果体现的并非单一 TLB 容量边界，而是现代处理器在多级 TLB 与虚拟化环境下，地址转换系统随工作集规模变化的整体性能特征。
