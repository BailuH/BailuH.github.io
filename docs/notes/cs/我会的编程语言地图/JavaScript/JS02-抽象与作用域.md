### 箭头函数
```js
// 基本形式
() => expression                    // 无参数，表达式体
param => expression                 // 单参数，表达式体（可省略括号）
(param) => expression               // 单参数，表达式体（带括号）
(param1, paramN) => expression      // 多参数，表达式体

() => { statements }                // 无参数，块体
param => { statements }             // 单参数，块体
(param1, paramN) => { statements }  // 多参数，块体

// 剩余参数
(a, b, ...r) => expression

// 默认参数
(a = 400, b = 20, c) => expression

// 解构赋值
([a, b] = [10, 20]) => expression
({ a, b } = { a: 10, b: 20 }) => expression

// async 箭头函数
async param => expression
async (param1, param2) => { statements }

// 传统匿名函数
(function (a) {
  return a + 100;
});

// 步骤 1：移除 function，添加箭头
(a) => {
  return a + 100;
};

// 步骤 2：移除 {} 和 return（隐式返回）
(a) => a + 100;

// 步骤 3：单参数可省略括号
a => a + 100;
```

### 函数
#### 定义与调用
JS的函数是一个很复杂的对象，编码的时候我们可以将其看作和C语言类似的，是一组代码的集合，用`function`关键字标记。但实际上函数又属于一个对象，更具体地说，是一个JS的标准内置对象（[查看MDN关于Function的解释](https://developer.mozilla.org/zh-CN/docs/Web/JavaScript/Reference/Global_Objects/Function)），存在属性和方法。更重要的是可以==像一个值==一样被传来传去。
```js
console.log(square(5)); // 25
function square(n) {
  return n * n;
} // 函数提升

// 利用函数表达式来创建函数
const square = function (number) {
  return number * number;
};
console.log(square(4)); // 16
```

但是和C语言严格的==先声明后调用==不同的是，
1. JS的函数声明存在*函数提升*现象，运行JS程序的时候，JS解释器会将整个函数声明提升到==当前作用域==的顶部。
2. 由于函数是一个对象，也可以被赋值给一个变量。这样的行为我们称为称为==函数表达式==（被视作一种特别的JS表达式），此时 `function` 关键字后可以省略函数名，通过变量来调用函数也可以，称为==匿名函数==。*不过匿名函数自然无法实现递归函数定义。*

C语言的函数参数传递分为值传递和地址传递两类，主要体现在**参数在函数作用域内的变化是否对外可见**。而在JS中的函数统一是==值传递==，但是有特别的地方：
1. 如果参数属于 `primitive` 不可变类型，那么参数在函数作用域内的变化对外就不可见。
2. 如果参数属于 `Object` 对象（包含一般对象、数组等），那么参数在函数作用域内的变化对外可见。
#### 使用`arguments`对象
函数的实际参数会被保存在一个类似数组的 `arguments` 对象中。在函数内，你可以通过`arguments[i]`使用传入的参数（*这里的参数可以比声明的要更多*)。
```js
function myConcat(separator) {
  let result = ""; // 初始化列表
  // 迭代 arguments
  for (let i = 1; i < arguments.length; i++) {
    result += arguments[i] + separator;
  }
  return result;
}

console.log(myConcat("、", "红", "橙", "蓝"));
// "红、橙、蓝、"
console.log(myConcat("；", "大象", "长颈鹿", "狮子", "猎豹"));
// "大象；长颈鹿；狮子；猎豹；"
console.log(myConcat("。", "智者", "罗勒", "牛至", "胡椒", "香菜"));
// "智者。罗勒。牛至。胡椒。香菜。"
```

### JS的作用域机制
#### 作用域概述
##### 作用域基本概念
- **函数作用域**：在函数内定义的变量只能在函数内部访问，外部无法访问
- **全局作用域**：定义在全局域中的函数可以访问所有定义在全局域中的变量
- **嵌套作用域**：在另一个函数中定义的函数可以访问其父函数中定义的所有变量和父函数有权访问的任何其他变量
##### 作用域链
- 当存在多层嵌套函数时，形成**作用域链**
- 链的第一个元素是最里面的作用域，最后一个元素是最外层的作用域   
- 内部函数可以访问外部函数的所有变量，但外部函数不能访问内部函数的变量
```javascript
// 全局作用域
const num1 = 20;
const num2 = 3;
const name = "Chamakh";

// 全局函数可以访问全局变量
function multiply() {
  return num1 * num2;  // 可以访问全局的 num1 和 num2
}

// 嵌套函数示例
function getScore() {
  const num1 = 2;  // 局部变量，遮蔽全局 num1
  const num2 = 3;  // 局部变量，遮蔽全局 num2
  
  function add() {
    return `${name} 的得分为 ${num1 + num2}`;  // 访问 getScore 的局部变量和全局的 name
  }
  
  return add();
}
```
##### 命名冲突与遮蔽
- 当同一作用域链中存在同名变量时，**更近的作用域有更高的优先权**
- 最近的优先级最高，最远的优先级最低
```js
function outside() {
  const x = 5;
  function inside(x) {  // 参数 x 遮蔽了外部的 x
    return x * 2;
  }
  return inside;
}

console.log(outside()(10)); // 20（使用 inside 的参数 x=10，而非 outside 的 x=5）
```
#### 函数递归
##### 递归的定义
- **递归函数**：调用自身的函数  
- 递归近似于循环，两者都重复执行相同代码，都需要**终止条件**避免无限递归
##### 递归调用的三种方式
在函数体内，可以通过以下等价方式调用自身：
1. **函数名**（如 `bar()`）
2. **`arguments.callee`**（已弃用，不推荐）
3. **作用域内指向该函数的变量名**（如 `foo()`）
```js
const foo = function bar() {
  // 以下调用等价：
  // bar()
  // arguments.callee()
  // foo()
};

// 遍历树结构（DOM）
function walkTree(node) {
  if (node === null) {
    return;  // 终止条件
  }
  // 对当前节点进行操作
  for (let i = 0; i < node.childNodes.length; i++) {
    walkTree(node.childNodes[i]);  // 递归遍历子节点
  }
}
```
#### 函数嵌套与闭包
##### 嵌套函数
- 可以在一个函数内部嵌套另一个函数
- 嵌套（内部）函数对其容器（外部）函数是**私有的**
- 内部函数只可以在外部函数中访问
```js
function addSquares(a, b) {
  function square(x) {  // 嵌套函数，只能在 addSquares 内部访问
    return x * x;
  }
  return square(a) + square(b);
}

console.log(addSquares(2, 3)); // 13
// console.log(square(2));  // Error: square is not defined
```
##### 闭包
###### 闭包定义
- **闭包**是可以拥有独立变量以及绑定了这些变量的环境（"封闭"了表达式）的表达式（通常是函数）
- 内部函数包含外部函数的作用域，形成闭包
- 当内部函数以某种方式被外部函数之外的任何作用域访问时，就创建了闭包
######  闭包的核心特性(重点！！)
1. **访问权限**：内部函数可以访问外部函数的参数和变量
2. **封装性**：外部函数不能使用内部函数的参数和变量
3. **变量保持**：闭包必须保存它可见作用域中所有参数和变量，即使外部函数已执行完毕
```js
function outside(x) {
  function inside(y) {
    return x + y;  // 访问外部函数的参数 x
  }
  return inside;  // 返回内部函数，形成闭包
}

const fnInside = outside(3);  // fnInside 是一个闭包，保存了 x=3
console.log(fnInside(5));     // 8（3 + 5）
console.log(outside(3)(5));   // 8
```
###### 多层嵌套与作用域链
- 函数可以多层嵌套（A 包含 B，B 包含 C）
- 每个内部函数都形成闭包，可以访问所有外层函数的变量
- 形成**作用域链**：C → B → A → 全局
```js
function A(x) {
  function B(y) {
    function C(z) {
      console.log(x + y + z);  // 可以访问 A 的 x、B 的 y 和自身的 z
    }
    C(3);
  }
  B(2);
}
A(1); // 输出 6（1 + 2 + 3）
```
