docker ps       # 列出所有正在运行容器

-a 参数           # 列出已停止容器



docker stop <容器ID或者容器名>     # 停止已经运行的容器



docker run <镜像名>       # 根据镜像创建容器并且启动容器

--name <容器名>             # 参数指定容器名

-p <8080:80>                   # 参数 指定映射端口（将主机8080端口映射到容器的80端口

-d 参数                               # 表示后台运行

-rm 参数                            # 容器推出后自动删除



docker start <容器名>      # 运行已有的容器



docker rm <容器名>         # 删除已有容器





## 镜像管理

##### docker pull <仓库名:镜像名>      # 拉取镜像

##### docker image ls                                                # 列出镜像

##### docker rmi <镜像名>\[:TAG]：                              # 删除镜像（若镜像被容器使用需先删除容器或加 -f）

##### docker commit <容器ID> <镜像名>                           # 将一个容器保存为镜像
